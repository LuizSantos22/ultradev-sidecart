<?php
/** @var UltraDev_SideCart_Block_Cart $this */
?>
<div id="ultradev-sidecart" class="usc-sidecart" style="display:none;">
    <div class="usc-overlay" id="usc-overlay"></div>
    <div class="usc-container" id="usc-container">
        <div class="usc-loading"><?php echo $this->__('Loading...') ?></div>
    </div>
</div>

<script type="text/javascript">
    if (typeof jQuery === 'undefined') {
        var script = document.createElement('script');
        script.src = '//ajax.googleapis.com/ajax/libs/jquery/3.6.0/jquery.min.js';
        script.onload = function() {
            jQuery.noConflict();
            initUltraDevSideCart();
        };
        document.head.appendChild(script);
    } else {
        jQuery.noConflict();
        initUltraDevSideCart();
    }

    function initUltraDevSideCart() {
        jQuery(document).ready(function($) {
            $.UltraDevSideCart = {
                config: {
                    cartUrl: '<?php echo Mage::getUrl('ultradev_sidecart/cart/index') ?>',
                    addToCartSelector: '.btn-cart, .actions .btn-cart',
                    cartLinkSelector: '.top-link-cart, .header-minicart .cart-link'
                },
                init: function() {
                    this.bindCartLink();
                    this.bindAddToCart();
                },
                bindCartLink: function() {
                    $(this.config.cartLinkSelector).on('click', function(e) {
                        e.preventDefault();
                        $.UltraDevSideCart.open();
                    });
                },
                bindAddToCart: function() {
                    $(document).on('click', this.config.addToCartSelector, function(e) {
                        var $btn = $(this);
                        var $form = $btn.closest('form');
                        if (!$form.length || $form.attr('id') !== 'product_addtocart_form') {
                            // For simple add-to-cart links (e.g., category list)
                            var url = $btn.attr('onclick');
                            if (url && url.match(/setLocation\('([^']+)'\)/)) {
                                url = RegExp.$1;
                                $.UltraDevSideCart.addToCartByUrl(url);
                                e.preventDefault();
                                return false;
                            }
                        }
                        // Product page form
                        if ($form.length && typeof productAddToCartForm !== 'undefined') {
                            e.preventDefault();
                            if (productAddToCartForm.validator.validate()) {
                                var formData = $form.serialize();
                                $.UltraDevSideCart.addToCart($form.attr('action'), formData);
                            }
                            return false;
                        }
                    });
                },
                addToCartByUrl: function(url) {
                    $.ajax({
                        url: url,
                        type: 'POST',
                        data: { isAjax: 1 },
                        dataType: 'json',
                        beforeSend: function() {
                            $.UltraDevSideCart.open(true);
                        },
                        success: function(resp) {
                            if (resp.status === 'success') {
                                $.UltraDevSideCart.updateCartContent(resp.payload);
                            } else {
                                alert(resp.message || 'Error adding to cart');
                            }
                        },
                        error: function() {
                            alert('Communication error');
                        }
                    });
                },
                addToCart: function(url, formData) {
                    $.ajax({
                        url: url,
                        type: 'POST',
                        data: formData + '&isAjax=1',
                        dataType: 'json',
                        beforeSend: function() {
                            $.UltraDevSideCart.open(true);
                        },
                        success: function(resp) {
                            if (resp.status === 'success') {
                                $.UltraDevSideCart.updateCartContent(resp.payload);
                            } else {
                                alert(resp.message || 'Error adding to cart');
                            }
                        },
                        error: function() {
                            alert('Communication error');
                        }
                    });
                },
                open: function(loading) {
                    $('#ultradev-sidecart').fadeIn(200);
                    $('body').addClass('usc-open');
                    if (loading) {
                        this.loadCartContent();
                    }
                },
                close: function() {
                    $('#ultradev-sidecart').fadeOut(200);
                    $('body').removeClass('usc-open');
                },
                loadCartContent: function() {
                    var $container = $('#usc-container');
                    $container.html('<div class="usc-loading">Loading...</div>');
                    $.ajax({
                        url: this.config.cartUrl,
                        type: 'GET',
                        dataType: 'json',
                        success: function(resp) {
                            if (resp.status === 'success') {
                                $.UltraDevSideCart.updateCartContent(resp.payload);
                            } else {
                                $container.html('<div class="usc-error">Error loading cart</div>');
                            }
                        },
                        error: function() {
                            $container.html('<div class="usc-error">Error loading cart</div>');
                        }
                    });
                },
                updateCartContent: function(payload) {
                    var $container = $('#usc-container');
                    if (!payload.items || payload.items.length === 0) {
                        $container.html($('#usc-empty-template').html());
                        return;
                    }
                    // Build cart HTML from payload (or reuse existing template via AJAX)
                    // For simplicity, reload the cart via AJAX again but with rendered HTML endpoint?
                    // Better: create an endpoint that returns rendered cart.phtml
                    // But we can also render client-side. Let's do a separate AJAX call to get rendered HTML.
                    $.ajax({
                        url: '<?php echo Mage::getUrl('ultradev_sidecart/cart/index') ?>?format=html',
                        type: 'GET',
                        success: function(html) {
                            $container.html(html);
                            $.UltraDevSideCart.bindCartEvents();
                        }
                    });
                },
                bindCartEvents: function() {
                    $('#usc-close, .usc-overlay').off('click').on('click', function() {
                        $.UltraDevSideCart.close();
                    });
                    $('#usc-continue').off('click').on('click', function() {
                        $.UltraDevSideCart.close();
                    });
                    // Quantity buttons
                    $('.usc-plus, .usc-minus').off('click').on('click', function() {
                        var $btn = $(this);
                        var $input = $btn.siblings('.usc-qty');
                        var itemId = $input.data('item-id');
                        var qty = parseInt($input.val());
                        if ($btn.hasClass('usc-plus')) qty++;
                        else if (qty > 1) qty--;
                        $input.val(qty);
                        $.UltraDevSideCart.updateItem(itemId, qty);
                    });
                    $('.usc-qty').off('change').on('change', function() {
                        var $input = $(this);
                        var itemId = $input.data('item-id');
                        var qty = parseInt($input.val());
                        if (isNaN(qty) || qty < 1) qty = 1;
                        $input.val(qty);
                        $.UltraDevSideCart.updateItem(itemId, qty);
                    });
                    $('.usc-remove').off('click').on('click', function() {
                        var itemId = $(this).data('item-id');
                        if (confirm(window.UscConfig.confirmRemove)) {
                            $.UltraDevSideCart.removeItem(itemId);
                        }
                    });
                    $('.usc-coupon-apply').off('click').on('click', function() {
                        var code = $('#usc-coupon-code').val();
                        $.UltraDevSideCart.applyCoupon(code);
                    });
                    $('.usc-coupon-remove').off('click').on('click', function() {
                        $.UltraDevSideCart.removeCoupon();
                    });
                },
                updateItem: function(itemId, qty) {
                    $.ajax({
                        url: window.UscConfig.urlUpdateItem,
                        type: 'POST',
                        data: { id: itemId, qty: qty, form_key: window.UscConfig.formKey },
                        dataType: 'json',
                        success: function(resp) {
                            if (resp.status === 'success') {
                                $.UltraDevSideCart.updateCartContent(resp.payload);
                            } else {
                                alert(resp.message);
                            }
                        }
                    });
                },
                removeItem: function(itemId) {
                    $.ajax({
                        url: window.UscConfig.urlRemoveItem,
                        type: 'POST',
                        data: { id: itemId, form_key: window.UscConfig.formKey },
                        dataType: 'json',
                        success: function(resp) {
                            if (resp.status === 'success') {
                                $.UltraDevSideCart.updateCartContent(resp.payload);
                            } else {
                                alert(resp.message);
                            }
                        }
                    });
                },
                applyCoupon: function(code) {
                    $.ajax({
                        url: window.UscConfig.urlApplyCoupon,
                        type: 'POST',
                        data: { coupon_code: code, form_key: window.UscConfig.formKey },
                        dataType: 'json',
                        success: function(resp) {
                            if (resp.status === 'success') {
                                $.UltraDevSideCart.updateCartContent(resp.payload);
                            } else {
                                $('.usc-coupon-msg').text(resp.message).fadeIn().delay(3000).fadeOut();
                            }
                        }
                    });
                },
                removeCoupon: function() {
                    $.ajax({
                        url: window.UscConfig.urlRemoveCoupon,
                        type: 'POST',
                        data: { form_key: window.UscConfig.formKey },
                        dataType: 'json',
                        success: function(resp) {
                            if (resp.status === 'success') {
                                $.UltraDevSideCart.updateCartContent(resp.payload);
                            }
                        }
                    });
                }
            };
            $.UltraDevSideCart.init();
        });
    }
</script>

            <a href="<?php echo $this->getUrl() ?>"><?php echo $this->__('Continue Shopping') ?></a>
        </div>
    </div>
</script>
